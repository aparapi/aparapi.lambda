80 template <typename T> void List<T>::add(T _value, int _line, char *_fileName){
81  Ref<T> *handle = new Ref<T>(_value, _line, _fileName);
82  handle->next = head;
83  head = handle;
84  count++;
85 }

87 template <typename T> void List<T>::remove(T _value, int _line, char *_fileName){
88  for (Ref<T> *ptr = head, *last=NULL; ptr != NULL; last=ptr, ptr = ptr->next){
89  if (ptr->value == _value){
90  if (last == NULL){ // head
91  head = ptr->next;
92  }else{ // !head
93  last->next = ptr->next;
94  }
95  delete ptr;
96  count--;
97  return;
98  }
99  }
100  fprintf(stderr, "FILE %s LINE %d failed to find %s to remove %0lx\n", _fileName, _line, name, (unsigned long)_value);
101 }

103 template <typename T> void List<T>::report(FILE *stream){
104  if (head != NULL){
105  fprintf(stream, "Resource report %d resources of type %s still in play ", count, name);
106  for (Ref<T> *ptr = head; ptr != NULL; ptr = ptr->next){
107  fprintf(stream, " %0lx(%d)", (unsigned long)ptr->value, ptr->line);
108  }
109  fprintf(stream, "\n");
110  }
