197  template<typename jT>
198  static void setInstanceField(JNIEnv* jenv, jobject instance, const char *fieldName, const char* signature, jT value) {
199  try {
200  jclass theClass = jenv->GetObjectClass(instance);
201  if (theClass == NULL || jenv->ExceptionCheck())
202  throw "bummer! getting class from instance\n";
203  jfieldID fieldId = jenv->GetFieldID(theClass,fieldName,"I");
204  if (fieldId == NULL || jenv->ExceptionCheck())
205  throw "bummer getting "+ getType(value) + "field '" + fieldName + "' \n";
206  setField(jenv, instance, fieldId, &value);
207  if (jenv->ExceptionCheck())
208  throw "bummer setting "+ getType(value) + "field '" + fieldName + "' \n";
209  } catch(std::string& se) {
210  jenv->ExceptionDescribe();
211  jenv->ExceptionClear();
212  fprintf(stderr,"%s",se.c_str());
213  }
214  }
